   preco <- (10)
   total <- (quantidade * preco)

   se(quantidade <= 5)entao
      valorFinal <- total * (100-2)/100 //ou *0.98
   fimse

      valorFinal <- total * (100-3)/100 //ou *0.97
   fimse

      valorFinal <- total * (100-5)/100 //ou *0.95
   fimse
